Count characters in your string

                function count (string) {  
                    obj = {};
                      if (string == ''){
                        return obj;
                      }
                      for (i = 0; i < string.length; i++) {
                        let key = string[i];
                        if (key in obj) {
                          obj[string[i]] = obj[string[i]] + 1;
                        }
                        else {
                        obj[string[i]] = 1;
                        }
                      }
                    return obj;
                  }

Square Every Digit

                function squareDigits(num){
                    res = String(num);
                   str = '';
                   for (i = 0; i < res.length; i++){
                     str = str + (res[i] * res[i]);
                   }
                   str = Number(str);
                   return str;
                   
                 }
